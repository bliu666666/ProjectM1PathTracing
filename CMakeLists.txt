project(RayTracer)


set(CMAKE_CXX_STANDARD 17)

    ${SRC_DIR}/position.cpp
    ${SRC_DIR}/ray.cpp

# Inclure les headers

    ${SRC_DIR}
    ${SHAPES_DIR}

)
